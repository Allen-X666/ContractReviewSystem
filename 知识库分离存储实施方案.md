# 知识库分离存储实施方案

## 一、背景与目标

### 1.1 现状分析

当前系统只有一个知识库——**法律文档知识库**，其核心架构如下：

| 组件 | 当前实现 |
|------|----------|
| 向量数据库 | ChromaDB（PersistentClient） |
| 持久化目录 | `合同审查/data/vector_db/law_documents/` |
| Collection 命名 | `law_documents_{document_type}`（如 `law_documents_labor`） |
| 检索逻辑 | `search_law_documents()` 遍历 `law_documents/` 下所有 Collection |
| 上传服务 | `LawUploadService` — 仅处理法律文档 |
| Agent 工具 | `search_contract_knowledge` → 调用 `get_law_retriever()` |
| 前端页面 | 知识库页面 (`/knowledge`) + 管理后台法律文档上传 (`/admin`) |

### 1.2 目标

建立两个独立的知识库，**分开存储、独立检索**：

| 知识库 | 用途 | 状态 |
|--------|------|------|
| 法律文档知识库 | 存储法律法规、法条条款 | ✅ 已完成 |
| 系统操作知识库 | 存储系统使用手册、操作指南、FAQ | ❌ 待建设 |

```python
def get_system_vector_stores() -> List[ChromaVectorStore]:
    """
    获取所有系统操作文档向量存储实例

    扫描 system_operations 持久化目录下的所有 Chroma 集合，
    返回可用的 ChromaVectorStore 列表。
    """
    global _system_stores_cache
    if _system_stores_cache is not None:
        return _system_stores_cache

    stores = []
    persist_dir = SYSTEM_VECTOR_DB_DIR

    if not os.path.exists(persist_dir):
        logger.warning(f"系统操作向量库目录不存在: {persist_dir}")
        return stores

    try:
        import chromadb
        from chromadb.config import Settings as ChromaSettings

        client = chromadb.PersistentClient(
            path=persist_dir,
            settings=ChromaSettings(anonymized_telemetry=False),
        )
        collections = client.list_collections()
        logger.info(f"发现 {len(collections)} 个系统操作 Chroma 集合")

        for collection in collections:
            name = collection.name
            count = collection.count()
            if count > 0:
                store = ChromaVectorStore(
                    collection_name=name,
                    persist_directory=persist_dir,
                )
                stores.append(store)
                logger.info(f"  加载系统操作集合: {name} ({count} 条记录)")
    except Exception as e:
        logger.error(f"扫描系统操作向量库失败: {e}")

    _system_stores_cache = stores
    return stores
```

#### 4.1.3 新增系统操作文档搜索函数

```python
def search_system_documents(
    query_embedding: List[float],
    top_k: int = 5,
) -> List[Tuple[ClauseVectorRecord, float]]:
    """
    在所有系统操作文档集合中搜索，合并结果按相似度排序返回
    """
    stores = get_system_vector_stores()
    if not stores:
        logger.warning("没有可用的系统操作文档向量库")
        return []

    all_results = []
    for store in stores:
        try:
            results = store.search(query_embedding, top_k=top_k)
            all_results.extend(results)
        except Exception as e:
            logger.error(f"搜索系统操作集合 {store.collection_name} 失败: {e}")

    all_results.sort(key=lambda x: x[1], reverse=True)
    return all_results[:top_k]
```

```python
class SystemDocumentRetriever(BaseRetriever):
    """
    系统操作文档检索器

    在所有已上传的系统操作文档向量库中进行相似度搜索。
    """

    def __init__(self, embeddings: Embeddings):
        self.embeddings = embeddings

    def retrieve(
        self,
        query: str,
        top_k: int = 5,
        filter_dict: Optional[Dict[str, Any]] = None,
    ) -> List[RetrievedClause]:
        logger.info(f"【系统操作检索】查询: '{query[:100]}...' | top_k={top_k}")

        query_embedding = self.embeddings.embed_query(query)
        vector_norm = sum(x ** 2 for x in query_embedding) ** 0.5
        if vector_norm < 1e-6:
            logger.error("【系统操作检索】查询向量为零向量")
            return []

        from .vector_store import search_system_documents
        results = search_system_documents(query_embedding, top_k=top_k)

        if not results:
            logger.warning("【系统操作检索】未找到任何结果")
            return []

        clauses = []
        for i, (record, score) in enumerate(results):
            clause = RetrievedClause.from_record(record, score)
            clauses.append(clause)
            logger.info(
                f"【系统操作检索】结果 {i+1}: "
                f"条款[{clause.clause_no}] 相似度={score:.4f} | "
                f"来源={clause.metadata.get('source', '未知')} | "
                f"内容: {clause.clause_content[:80]}..."
            )

        logger.info(f"【系统操作检索】共返回 {len(clauses)} 条结果")
        return clauses
```

#### 4.2.2 新增 CachedSystemDocumentRetriever

参考 `CachedLawDocumentRetriever`，新增带缓存的系统操作检索器：

```python
_cached_system_retriever: Optional[CachedSystemDocumentRetriever] = None


class CachedSystemDocumentRetriever(BaseRetriever):
    """
    带缓存的系统操作文档检索器
    """

    def __init__(
        self,
        embeddings: Optional[Embeddings] = None,
        cache_size: int = 1000,
        cache_ttl: int = 3600,
    ):
        self._embeddings = embeddings
        self._cache = SimpleTTLCache(maxsize=cache_size, ttl=cache_ttl)
        self._embedding_lock = asyncio.Lock()
        self._stats = {
            "total_queries": 0,
            "cache_hits": 0,
            "cache_misses": 0,
        }

    async def _get_embeddings(self) -> Embeddings:
        if self._embeddings is None:
            async with self._embedding_lock:
                if self._embeddings is None:
                    self._embeddings = get_embeddings(
                        embedding_type=settings.EMBEDDING_TYPE,
                        model=settings.EMBEDDING_MODEL,
                    )
        return self._embeddings

    def _generate_cache_key(self, query: str, top_k: int) -> str:
        import hashlib
        key_content = f"system:{query.strip().lower()}:{top_k}"
        return hashlib.md5(key_content.encode()).hexdigest()

    async def retrieve(
        self,
        query: str,
        top_k: int = 5,
        filter_dict: Optional[Dict[str, Any]] = None,
        use_cache: bool = True,
    ) -> List[RetrievedClause]:
        self._stats["total_queries"] += 1

        if filter_dict:
            use_cache = False

        cache_key = self._generate_cache_key(query, top_k)

        if use_cache:
            cached_result = await self._cache.get(cache_key)
            if cached_result is not None:
                self._stats["cache_hits"] += 1
                return cached_result
            self._stats["cache_misses"] += 1

        embeddings = await self._get_embeddings()
        query_embedding = embeddings.embed_query(query)

        from .vector_store import search_system_documents
        results = search_system_documents(query_embedding, top_k=top_k)

        if not results:
            return []

        clauses = [
            RetrievedClause.from_record(record, score)
            for record, score in results
        ]

        if use_cache:
            await self._cache.set(cache_key, clauses)

        return clauses
```

```python
"""
系统操作文档上传异步处理服务
"""

import uuid
import os
from datetime import datetime
from typing import Dict, Optional, Any
from threading import Lock, Thread
from queue import Queue, Empty
import time

from 合同审查.app.core.config import settings

SYSTEM_VECTOR_DB_DIR = os.path.join(
    os.path.dirname(__file__), "..", "..", "data", "vector_db", "system_operations"
)
os.makedirs(SYSTEM_VECTOR_DB_DIR, exist_ok=True)


class SystemOpsUploadTask:
    """系统操作文档上传任务"""

    def __init__(
        self,
        task_id: str,
        file_name: str,
        file_content: bytes,
        file_ext: str,
        category: str,
        description: Optional[str] = None,
    ):
        self.task_id = task_id
        self.file_name = file_name
        self.file_content = file_content
        self.file_ext = file_ext
        self.category = category
        self.description = description

        self.status = "pending"
        self.progress = 0
        self.message = "任务等待中"
        self.result: Optional[Dict[str, Any]] = None
        self.error: Optional[str] = None

        self.created_at = datetime.now()
        self.started_at: Optional[datetime] = None
        self.completed_at: Optional[datetime] = None

    def to_dict(self) -> Dict[str, Any]:
        return {
            "task_id": self.task_id,
            "file_name": self.file_name,
            "category": self.category,
            "status": self.status,
            "progress": self.progress,
            "message": self.message,
            "result": self.result,
            "error": self.error,
            "created_at": self.created_at.isoformat() if self.created_at else None,
            "started_at": self.started_at.isoformat() if self.started_at else None,
            "completed_at": self.completed_at.isoformat() if self.completed_at else None,
        }


class SystemOpsUploadService:
    """系统操作文档上传异步处理服务"""

    _instance = None
    _lock = Lock()

    def __new__(cls):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super().__new__(cls)
                    cls._instance._initialized = False
        return cls._instance

    def __init__(self):
        if self._initialized:
            return
        self._initialized = True
        self.tasks: Dict[str, SystemOpsUploadTask] = {}
        self.task_queue: Queue = Queue()
        self.tasks_lock = Lock()
        self.document_loader = DocumentLoader()
        self.worker_thread: Optional[Thread] = None
        self.running = False

    def start(self):
        if not self.running:
            self.running = True
            self.worker_thread = Thread(target=self._process_queue, daemon=True)
            self.worker_thread.start()
            logger.info("系统操作文档上传服务已启动")

    def stop(self):
        self.running = False
        if self.worker_thread:
            self.worker_thread.join(timeout=5)

    def create_task(
        self,
        file_name: str,
        file_content: bytes,
        file_ext: str,
        category: str,
        description: Optional[str] = None,
    ) -> str:
        task_id = str(uuid.uuid4())
        task = SystemOpsUploadTask(
            task_id=task_id,
            file_name=file_name,
            file_content=file_content,
            file_ext=file_ext,
            category=category,
            description=description,
        )
        with self.tasks_lock:
            self.tasks[task_id] = task
        self.task_queue.put(task_id)
        logger.info(f"创建系统操作文档上传任务: {task_id}, 文件名: {file_name}")
        return task_id

    def get_task(self, task_id: str) -> Optional[SystemOpsUploadTask]:
        with self.tasks_lock:
            return self.tasks.get(task_id)

    def _process_queue(self):
        while self.running:
            try:
                task_id = self.task_queue.get(timeout=1)
                self._process_task(task_id)
            except Empty:
                continue
            except Exception as e:
                if self.running:
                    logger.error(f"处理系统操作文档队列时出错: {e}", exc_info=True)
                time.sleep(1)

    def _process_task(self, task_id: str):
        """处理单个系统操作文档上传任务"""
        with self.tasks_lock:
            task = self.tasks.get(task_id)
            if not task:
                return

        task.status = "processing"
        task.started_at = datetime.now()
        task.message = "正在处理文档"
        task.progress = 10

        try:
            # 1. 加载文档
            task.message = "正在加载文档内容"
            task.progress = 20

            file_ext = task.file_ext.lower()
            chunks = self.document_loader.load(
                file_obj=__import__('io').BytesIO(task.file_content),
                file_name=task.file_name,
            )

            if not chunks:
                raise ValueError("文档解析结果为空")

            # 2. 为每个 chunk 添加系统操作元数据
            unique_filename = uuid.uuid4().hex
            for i, chunk in enumerate(chunks):
                chunk.clause_id = f"sysops_{unique_filename}_chunk_{i}"
                chunk.metadata.update({
                    "knowledge_base": "system_operations",
                    "category": task.category,
                    "description": task.description or "",
                    "upload_time": datetime.now().isoformat(),
                })

            task.message = "正在生成向量嵌入"
            task.progress = 50

            # 3. 生成向量
            embeddings = get_embeddings(
                embedding_type=settings.EMBEDDING_TYPE,
                model=settings.EMBEDDING_MODEL,
            )
            texts = [chunk.clause_content for chunk in chunks]
            clause_embeddings = embeddings.embed_documents(texts)

            task.message = "正在存入向量数据库"
            task.progress = 80

            # 4. 存入系统操作专用向量库
            collection_name = f"system_ops_{task.category}"
            vector_store = ChromaVectorStore(
                collection_name=collection_name,
                persist_directory=SYSTEM_VECTOR_DB_DIR,
            )

            ids = vector_store.add_clauses(
                chunks=chunks,
                embeddings=clause_embeddings,
            )

            # 5. 完成
            task.status = "completed"
            task.progress = 100
            task.completed_at = datetime.now()
            task.message = "文档处理完成"
            task.result = {
                "task_id": task.task_id,
                "file_name": task.file_name,
                "category": task.category,
                "chunks_count": len(chunks),
                "embeddings_count": len(clause_embeddings),
                "vector_ids": ids[:5] if len(ids) > 5 else ids,
            }

            # 清除缓存，确保下次检索能加载新数据
            _vector_store_module._system_stores_cache = None
            _retriever_module._system_retriever_cache = None
            if hasattr(_retriever_module, '_cached_system_retriever'):
                _retriever_module._cached_system_retriever = None

            logger.info(f"[{task_id}] 系统操作文档处理完成")

        except Exception as e:
            logger.error(f"[{task_id}] 处理系统操作文档失败: {str(e)}", exc_info=True)
            task.status = "failed"
            task.error = str(e)
            task.message = f"处理失败: {str(e)}"
            task.completed_at = datetime.now()
```

```python
@tool
def search_system_knowledge(query: str) -> str:
    """
    搜索系统操作知识库，查找系统使用相关的操作指南和帮助信息。
```

### 4.5 API 接口层改造

#### 4.5.1 新增系统操作文档上传接口

**文件**：`合同审查/app/api/v1/endpoints/laws.py`

在文件末尾新增：

```python
from 合同审查.app.services.system_ops_upload_service import system_ops_upload_service

# 启动系统操作文档上传服务
system_ops_upload_service.start()


@router.post("/system-ops/upload")
async def upload_system_ops_document(
    file: UploadFile = File(..., description="系统操作文档文件"),
    category: str = Form(..., description="文档分类：guide/faq/manual"),
    description: str = Form(None, description="文档说明"),
):
    """
    上传系统操作文档（异步处理）

    支持的文档分类：
    - guide: 操作指南
    - faq: 常见问题
    - manual: 用户手册
    """
    try:
        allowed_extensions = {".pdf", ".doc", ".docx", ".md", ".txt"}
        file_ext = os.path.splitext(file.filename)[1].lower()
        if file_ext not in allowed_extensions:
            return Result(code=400, message=f"不支持的文件格式，仅支持: {', '.join(allowed_extensions)}")

        file_content = await file.read()
        if len(file_content) > 50 * 1024 * 1024:
            return Result(code=400, message="文件大小超过限制（50MB）")

        task_id = system_ops_upload_service.create_task(
            file_name=file.filename,
            file_content=file_content,
            file_ext=file_ext,
            category=category,
            description=description,
        )

        return Result(
            code=200,
            message="系统操作文档上传任务已创建，正在后台处理",
            data={
                "task_id": task_id,
                "file_name": file.filename,
                "category": category,
                "status": "pending",
            }
        )
    except Exception as e:
        logger.error(f"创建系统操作文档上传任务失败: {str(e)}", exc_info=True)
        return Result(code=500, message=f"上传失败: {str(e)}")


@router.get("/system-ops/upload/{task_id}/status")
async def get_system_ops_upload_status(task_id: str):
    """获取系统操作文档上传任务状态"""
    try:
        task = system_ops_upload_service.get_task(task_id)
        if not task:
            return Result(code=404, message="任务不存在")
        return Result(code=200, message="获取任务状态成功", data=task.to_dict())
    except Exception as e:
        return Result(code=500, message=f"获取状态失败: {str(e)}")


@router.get("/system-ops/upload/tasks")
async def list_system_ops_upload_tasks():
    """获取所有系统操作文档上传任务列表"""
    try:
        tasks = []
        with system_ops_upload_service.tasks_lock:
            for task in system_ops_upload_service.tasks.values():
                tasks.append(task.to_dict())
        tasks.sort(key=lambda x: x.get("created_at", ""), reverse=True)
        return Result(code=200, message="获取任务列表成功", data={"total": len(tasks), "tasks": tasks})
    except Exception as e:
        return Result(code=500, message=f"获取列表失败: {str(e)}")
```

## 六、数据流对比

### 6.1 法律文档知识库数据流（已有）

```

    ↓

    ↓
后台线程处理：
  1. parse_law_content() 解析法律文件
  2. get_embeddings().embed_documents() 生成向量
  3. ChromaVectorStore(collection_name="law_documents_{type}",
                       persist_directory="data/vector_db/law_documents/")
     .add_clauses() 存入 ChromaDB
    ↓
Agent 调用 search_contract_knowledge()
    ↓
get_law_retriever() → LawDocumentRetriever
    ↓
search_law_documents() → 遍历 law_documents/ 下所有 Collection
    ↓
返回检索结果给 LLM 生成回答
```

```
管理员上传系统操作文档
    ↓
laws.py POST /admin/laws/system-ops/upload
    ↓
system_ops_upload_service.create_task()
    ↓
后台线程处理：
  1. DocumentLoader 加载文档
  2. get_embeddings().embed_documents() 生成向量
  3. ChromaVectorStore(collection_name="system_ops_{category}",

Agent 调用 search_system_knowledge()
    ↓
get_system_retriever() → SystemDocumentRetriever

search_system_documents() → 遍历 system_operations/ 下所有 Collection
    ↓
返回检索结果给 LLM 生成回答
```

## 九、注意事项

### 9.1 关键设计决策

1. **独立的 ChromaDB 实例**：两个知识库使用不同的 `persist_directory`，确保物理隔离，互不影响。
2. **Collection 命名前缀**：法律文档使用 `law_documents_` 前缀，系统操作使用 `system_ops_` 前缀，便于管理和识别。
3. **独立的缓存**：每个知识库的检索器拥有独立的缓存实例，避免缓存污染。
4. **独立的上传服务**：两个上传服务独立运行，各自维护任务队列和线程。

### 9.2 后续扩展

- 当需要新增第三种知识库时，按照相同模式新增即可：
